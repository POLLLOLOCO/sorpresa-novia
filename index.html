import tkinter as tk import random # ===== PERSONALIZA ===== NOMBRE = "Mi niña 💖" # ======================= ventana = tk.Tk() ventana.title("Para ti ❤️") ventana.geometry("600x550") ventana.config(bg="black") canvas = tk.Canvas(ventana, width=600, height=550, bg="black") canvas.pack() # FOTO 🖼️ (AJUSTADA) try: foto = tk.PhotoImage(file="foto.png") foto = foto.subsample(3, 3) # cambia a 2, 3 o 4 si quieres canvas.create_image(300, 120, image=foto) canvas.image = foto except Exception as e: print("Error cargando imagen:", e) # TEXTO 💌 (JUSTO DEBAJO DE LA FOTO) texto = canvas.create_text( 300, 450, # 👈 posición ajustada debajo de la imagen text="", fill="white", font=("Helvetica", 18, "bold"), width=500 ) # MENSAJES mensajes = [ f"Hola {NOMBRE} ❤️", "Quería hacer algo especial para ti...", "Eres lo mejor que me ha pasado 💖", "Gracias por estar siempre a mi lado 🥰", "Quiero que seas tú...", "La mujer con la que pase el resto de mi vida 💕", "no solo me ragalas felicidad", "sino que me haces una mejor persona", "quiero ser cada dia mejor para estar a tu lado", "ya sabes todo lo que quiero contigo, pero", "siempre me gusta recordarte lo maravillosa", "que eres, como mujer, persona, como constructo", "como lo que eres, gracias por existir y por ", "los moemntos que me has dejado pasar a tu lado", "y solo quiero que sepas que", "TE AMO ❤️❤️❤️ MUCHOOOOOOOOOOOOOOOOO" ] # EFECTO MÁQUINA DE ESCRIBIR def escribir(texto_completo, i=0): if i <= len(texto_completo): canvas.itemconfig(texto, text=texto_completo[:i]) ventana.after(40, escribir, texto_completo, i+1) else: ventana.after(1500, siguiente) indice = 0 def siguiente(): global indice indice += 1 if indice < len(mensajes): escribir(mensajes[indice]) else: final() # FINAL def final(): canvas.itemconfig(texto, text="¿Quieres seguir conmigo? 💍") boton_si = tk.Button(ventana, text="Sí ❤️", font=("Arial", 12), command=respuesta) boton_no = tk.Button(ventana, text="No 😢", font=("Arial", 12)) def mover(event): x = random.randint(50, 550) y = random.randint(350, 500) canvas.coords(btn_no, x, y) boton_no.bind("", mover) canvas.create_window(200, 520, window=boton_si) btn_no = canvas.create_window(400, 520, window=boton_no) # EXPLOSIÓN FINAL 💥 def explosion(): for _ in range(50): x = random.randint(0, 600) y = random.randint(0, 550) canvas.create_text(x, y, text="❤", fill="red", font=("Arial", 20)) # RESPUESTA def respuesta(): canvas.delete("all") canvas.create_text( 300, 250, text=f"Sabía que dirías que sí {NOMBRE} 😍\nTE AMO ❤️❤️❤️", fill="white", font=("Arial", 22) ) explosion() # INICIO 🎬 def intro(): canvas.itemconfig(texto, text="") escribir(mensajes[0]) def iniciar(): boton_inicio.destroy() ventana.after(500, intro) boton_inicio = tk.Button( ventana, text="Haz clic aquí ❤️", font=("Arial", 14), command=iniciar ) canvas.create_window(300, 450, window=boton_inicio) # CORAZONES ❤️ corazones = [] def crear_corazon(): x = random.randint(0, 600) c = canvas.create_text(x, 550, text="❤", fill="red", font=("Arial", 18)) corazones.append(c) def animar(): for c in corazones: canvas.move(c, 0, -3) if random.random() < 0.1: crear_corazon() ventana.after(50, animar) animar() ventana.mainloop()
